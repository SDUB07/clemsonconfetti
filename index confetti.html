  <meta charset="UTF-8">
  <title>Clemson Confetti</title>
  <script src="https://cdn.jsdelivr.net/npm/canvas-confetti@1.4.0/dist/confetti.browser.min.js"></script>
  <style>
    html, body {
      margin: 0;
      padding: 0;
      background: transparent;
      overflow: hidden;
      height: 100%;
    }
    canvas {
      position: fixed;
      top: 0;
      left: 0;
      pointer-events: none;
      z-index: 9999;
    }
  </style>
</head>
<body>
  <script>
    const colors = ['#F56600', '#522D80']; // Clemson colors

    const duration = 60 * 1000; // 1 minute
    const end = Date.now() + duration;

    (function frame() {
      confetti({
        particleCount: 2,
        startVelocity: 10,
        spread: 360,
        ticks: 300,
        gravity: 0.2,
        origin: {
          x: Math.random(),
          y: 0
        },
        colors: colors
      });

      if (Date.now() < end) {
        requestAnimationFrame(frame);
      }
    })();
  </script>
</body>
